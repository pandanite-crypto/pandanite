 cmake_minimum_required(VERSION 2.8.12)
 project(SimpleCoin)

 
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++17")
set (CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set (CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})

 include(${CMAKE_BINARY_DIR}/build/conanbuildinfo.cmake)

include_directories(/usr/local/include/)

file(GLOB CORE_SOURCES "src/core/*.cpp")
file(GLOB CRYPTO_SOURCES "src/external/ed25519/*.c")
file(GLOB MURMUR_SOURCES "src/external/murmurhash3/*.cpp")

 add_executable(tests ${CORE_SOURCES} ${CRYPTO_SOURCES} ${MURMUR_SOURCES} ./src/tests/tests.cpp)

 add_executable(benchmark ${CORE_SOURCES} ${CRYPTO_SOURCES} ${MURMUR_SOURCES} ./src/tools/benchmark.cpp)
 add_executable(restore ${CORE_SOURCES} ${CRYPTO_SOURCES} ${MURMUR_SOURCES} ./src/tools/restore.cpp)
 add_executable(txgen ${CORE_SOURCES} ${CRYPTO_SOURCES} ${MURMUR_SOURCES} ./src/tools/txgen.cpp)
 if (APPLE)
    target_link_libraries(tests ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.a /usr/local/Cellar/snappy/1.1.9/lib/libsnappy.a)

    target_link_libraries(benchmark ${CONAN_LIBS} /usr/local/Cellar/leveldb/1.23/lib/libleveldb.a /usr/local/Cellar/snappy/1.1.9/lib/libsnappy.a)
